# FUNCTION: Convert DMS to Decimal Degrees
# ============================================================================

def dms_to_decimal(dms_str):
    """
    Convert coordinates from DMS (Degrees, Minutes, Seconds) to decimal degrees.
    Example: 40В°58'02"N -> 40.967222

    Args:
        dms_str (str): Coordinate in DMS format (e.g., "40В°58'02\"N")

    Returns:
        float: Coordinate in decimal degrees
    """
    direction = dms_str[-1]
    dms_str = dms_str[:-1]

    parts = dms_str.replace('В°', ' ').replace("'", ' ').replace('"', '').split()

    degrees = float(parts[0])
    minutes = float(parts[1]) if len(parts) > 1 else 0
    seconds = float(parts[2]) if len(parts) > 2 else 0

    decimal = degrees + minutes/60 + seconds/3600

    if direction in ['S', 'W']:
        decimal = -decimal

    return decimal

    radius=30,
